package com.nopcommerce;

import java.time.Duration;

import org.openqa.selenium.By;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.edge.EdgeDriver;
import org.openqa.selenium.interactions.Actions;
import org.openqa.selenium.remote.service.DriverFinder;

public class Nopcommerce {

	public static void main(String[] args) {
		EdgeDriver driver=new EdgeDriver();
		driver.get("https://demo.nopcommerce.com/");
		driver.manage().window().maximize();
		driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(40));
		
		driver.findElement(By.linkText("Log in")).click();
		driver.findElement(By.name("Email")).sendKeys("tester1234@gmail.com");
		driver.findElement(By.id("Password")).sendKeys("tester@1234");
		driver.findElement(By.xpath("(//button[@type='submit'])[2]")).click();
		driver.findElement(By.id("small-searchterms")).sendKeys("iMac");
		driver.findElement(By.cssSelector("button[type='submit']")).click();
		
		driver.findElement(By.xpath("(//a[contains(text(),'Electronics')])[3]")).click();
		
		
		driver.findElement(By.xpath("(//*[contains(text(),'Cell phones ')])[3]")).click();
		driver.findElement(By.xpath("(//button[contains(.,'Add to cart')])[1]")).click();
        driver.findElement(By.xpath("(//*[contains(text(),'Shopping cart')])[1]")).click();
	}

}
